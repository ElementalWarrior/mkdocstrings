91class PythonCollector(BaseCollector): 

92 """The class responsible for loading Jinja templates and rendering them. 

93 

94 It defines some configuration options, implements the `render` method, 

95 and overrides the `update_env` method of the [`BaseRenderer` class][mkdocstrings.handlers.base.BaseRenderer]. 

96 """ 

97 

98 default_config: dict = {"filters": ["!^_[^_]"]} 

99 """The default selection options. 

100 

101 Option | Type | Description | Default 

102 ------ | ---- | ----------- | ------- 

103 **`filters`** | `List[str]` | Filter members with regular expressions. | `[ "!^_[^_]" ]` 

104 **`members`** | `Union[bool, List[str]]` | Explicitly select the object members. | *`pytkdocs` default: `True`* 

105 

106 If `members` is a list of names, filters are applied only on the members children (not the members themselves). 

107 If `members` is `False`, none are selected. 

108 If `members` is `True` or an empty list, filters are applied on all members and their children. 

109 

110 Members affect only the first layer of objects, while filters affect the whole object-tree recursively. 

111 

112 Every filters is run against every object name. An object can be un-selected by a filter and re-selected by the 

113 next one: 

114 

115 - `"!^_"`: exclude all objects starting with an underscore 

116 - `"^__"`: but select all objects starting with **two** underscores 

117 

118 Obviously one could use a single filter instead: `"!^_[^_]"`, which is the default. 

119 """

121 def __init__(self, setup_commands: Optional[List[str]] = None) -> None: 

122 """Initialize the object. 

123 

124 When instantiating a Python collector, we open a subprocess in the background with `subprocess.Popen`. 

125 It will allow us to feed input to and read output from this subprocess, keeping it alive during 

126 the whole documentation generation. Spawning a new Python subprocess for each "autodoc" instruction would be 

127 too resource intensive, and would slow down `mkdocstrings` a lot. 

128 

129 Arguments: 

130 setup_commands: A list of python commands as strings to be executed in the subprocess before `pytkdocs`. 

131 """ 

132 log.debug("Opening 'pytkdocs' subprocess") 

133 env = os.environ.copy() 

134 env["PYTHONUNBUFFERED"] = "1" 

135 

136 if setup_commands: 

137 # prevent the Python interpreter or the setup commands 

138 # from writing to stdout as it would break pytkdocs output 

139 commands = [ 

140 "import sys", 

141 "from io import StringIO", 

142 "from pytkdocs.cli import main as pytkdocs", 

143 "sys.stdout = StringIO()", # redirect stdout to memory buffer 

144 *setup_commands, 

145 "sys.stdout.flush()", 

146 "sys.stdout = sys.__stdout__", # restore stdout 

147 "pytkdocs(['--line-by-line'])", 

148 ] 

149 cmd = [sys.executable, "-c", "; ".join(commands)] 

150 else: 

151 cmd = [sys.executable, "-m", "pytkdocs", "--line-by-line"] 

152 

153 self.process = Popen( # noqa: S603,S607 (we trust the input, and we don't want to use the absolute path) 

154 cmd, 

155 universal_newlines=True, 

156 stderr=PIPE, 

157 stdout=PIPE, 

158 stdin=PIPE, 

159 bufsize=-1, 

160 env=env, 

161 )

163 def collect(self, identifier: str, config: dict) -> CollectorItem: 

164 """Collect the documentation tree given an identifier and selection options. 

165 

166 In this method, we feed one line of JSON to the standard input of the subprocess that was opened 

167 during instantiation of the collector. Then we read one line of JSON on its standard output. 

168 

169 We load back the JSON text into a Python dictionary. 

170 If there is a decoding error, we log it as error and raise a CollectionError. 

171 

172 If the dictionary contains an `error` key, we log it as error (with the optional `traceback` value), 

173 and raise a CollectionError. 

174 

175 If the dictionary values for keys `loading_errors` and `parsing_errors` are not empty, 

176 we log them as warnings. 

177 

178 Then we pick up the only object within the `objects` list (there's always only one, because we collect 

179 them one by one), rebuild it's categories lists 

180 (see [`rebuild_category_lists()`][mkdocstrings.handlers.python.rebuild_category_lists]), 

181 and return it. 

182 

183 Arguments: 

184 identifier: The dotted-path of a Python object available in the Python path. 

185 config: Selection options, used to alter the data collection done by `pytkdocs`. 

186 

187 Raises: 

188 CollectionError: When there was a problem collecting the object documentation. 

189 

190 Returns: 

191 The collected object-tree. 

192 """ 

193 final_config = ChainMap(config, self.default_config) 

194 

195 log.debug("Preparing input") 

196 json_input = json.dumps({"objects": [{"path": identifier, **final_config}]}) 

197 

198 log.debug("Writing to process' stdin") 

199 self.process.stdin.write(json_input + "\n") # type: ignore 

200 self.process.stdin.flush() # type: ignore 

201 

202 log.debug("Reading process' stdout") 

203 stdout = self.process.stdout.readline() # type: ignore 

204 

205 log.debug("Loading JSON output as Python object") 

206 try: 

207 result = json.loads(stdout) 

208 except json.decoder.JSONDecodeError as exception: 

209 error = "\n".join(("Error while loading JSON:", stdout, traceback.format_exc())) 

210 raise CollectionError(error) from exception 

211 

212 error = result.get("error") 

213 if error: 213 ↛ 214line 213 didn't jump to line 214, because the condition on line 213 was never true

214 if "traceback" in result: 

215 error += f"\n{result['traceback']}" 

216 raise CollectionError(error) 

217 

218 for loading_error in result["loading_errors"]: 218 ↛ 219line 218 didn't jump to line 219, because the loop on line 218 never started

219 log.warning(loading_error) 

220 

221 for errors in result["parsing_errors"].values(): 221 ↛ 222line 221 didn't jump to line 222, because the loop on line 221 never started

222 for parsing_error in errors: 

223 log.warning(parsing_error) 

224 

225 # We always collect only one object at a time 

226 result = result["objects"][0] 

227 

228 log.debug("Rebuilding categories and children lists") 

229 rebuild_category_lists(result) 

230 

231 return result

235 log.debug("Tearing process down") 

236 self.process.terminate()

266def rebuild_category_lists(obj: dict) -> None: 

267 """Recursively rebuild the category lists of a collected object. 

268 

269 Since `pytkdocs` dumps JSON on standard output, it must serialize the object-tree and flatten it to reduce data 

270 duplication and avoid cycle-references. Indeed, each node of the object-tree has a `children` list, containing 

271 all children, and another list for each category of children: `attributes`, `classes`, `functions`, `methods` 

272 and `modules`. It replaces the values in category lists with only the paths of the objects. 

273 

274 Here, we reconstruct these category lists by picking objects in the `children` list using their path. 

275 

276 For each object, we recurse on every one of its children. 

277 

278 Arguments: 

279 obj: The collected object, loaded back from JSON into a Python dictionary. 

280 """ 

281 for category in ("attributes", "classes", "functions", "methods", "modules"): 

282 obj[category] = [obj["children"][path] for path in obj[category]] 

283 obj["children"] = [child for _, child in obj["children"].items()] 

284 for child in obj["children"]: 

285 rebuild_category_lists(child)
