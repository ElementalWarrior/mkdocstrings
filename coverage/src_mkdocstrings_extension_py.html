1"""This module holds the code of the Markdown extension responsible for matching "autodoc" instructions. 

2 

3The extension is composed of a Markdown [block processor](https://python-markdown.github.io/extensions/api/#blockparser) 

4that matches indented blocks starting with a line like '::: identifier'. 

5 

6For each of these blocks, it uses a [handler][mkdocstrings.handlers.base.BaseHandler] to collect documentation about 

7the given identifier and render it with Jinja templates. 

8 

9Both the collection and rendering process can be configured by adding YAML configuration under the "autodoc" 

10instruction: 

11 

12```yaml 

13::: some.identifier 

14 handler: python 

15 selection: 

16 option1: value1 

17 option2: 

18 - value2a 

19 - value2b 

20 rendering: 

21 option_x: etc 

22``` 

23"""

50class AutoDocProcessor(BlockProcessor): 

51 """Our "autodoc" Markdown block processor. 

52 

53 It has a [`test` method][mkdocstrings.extension.AutoDocProcessor.test] that tells if a block matches a criterion, 

54 and a [`run` method][mkdocstrings.extension.AutoDocProcessor.run] that processes it. 

55 

56 It also has utility methods allowing to get handlers and their configuration easily, useful when processing 

57 a matched block. 

58 """

95 """Run code on the matched blocks. 

96 

97 The identifier and configuration lines are retrieved from a matched block 

98 and used to collect and render an object.

100 Arguments: 

101 parent: The parent element in the XML tree. 

102 blocks: The rest of the blocks to be processed. 

103 """

117 heading_level = match["heading"].count("#")

138 def _process_block(self, identifier: str, yaml_block: str, heading_level: int = 0) -> Tuple[str, Sequence[Element]]: 

139 """Process an autodoc block. 

140 

141 Arguments: 

142 identifier: The identifier of the object to collect and render. 

143 yaml_block: The YAML configuration. 

144 heading_level: Suggested level of the the heading to insert (0 to ignore). 

145 

146 Raises: 

147 PluginError: When something wrong happened during collection. 

148 TemplateNotFound: When a template used for rendering could not be found. 

149 

150 Returns: 

151 Rendered HTML and the list of heading elements encoutered. 

152 """ 

153 config = yaml.safe_load(yaml_block) or {} 

154 handler_name = self._handlers.get_handler_name(config) 

155 

156 log.debug(f"Using handler '{handler_name}'") 

157 handler_config = self._handlers.get_handler_config(handler_name) 

158 handler = self._handlers.get_handler(handler_name, handler_config) 

159 

160 selection, rendering = get_item_configs(handler_config, config) 

161 if heading_level: 161 ↛ 162line 161 didn't jump to line 162, because the condition on line 161 was never true

162 rendering = ChainMap(rendering, {"heading_level": heading_level}) # like setdefault 

163 

164 log.debug("Collecting data") 

165 try: 

166 data: CollectorItem = handler.collector.collect(identifier, selection) 

167 except CollectionError as exception: 

168 log.error(str(exception)) 

169 if PluginError is SystemExit: # When MkDocs 1.2 is sufficiently common, this can be dropped. 

170 log.error(f"Error reading page '{self._autorefs.current_page}':") 

171 raise PluginError(f"Could not collect '{identifier}'") from exception 

172 

173 if not self._updated_env: 

174 log.debug("Updating renderer's env") 

175 handler.renderer._update_env(self.md, self._config) # noqa: W0212 (protected member OK) 

176 self._updated_env = True 

177 

178 log.debug("Rendering templates") 

179 try: 

180 rendered = handler.renderer.render(data, rendering) 

181 except TemplateNotFound as exc: 

182 theme_name = self._config["theme_name"] 

183 log.error( 

184 f"Template '{exc.name}' not found for '{handler_name}' handler and theme '{theme_name}'.", 

185 ) 

186 raise 

187 

188 return (rendered, handler.renderer.get_headings())

191def get_item_configs(handler_config: dict, config: dict) -> Tuple[Mapping, Mapping]: 

192 """Get the selection and rendering configuration merged into the global configuration of the given handler. 

193 

194 Arguments: 

195 handler_config: The global configuration of a handler. It can be an empty dictionary. 

196 config: The configuration to merge into the global handler configuration. 

197 

198 Returns: 

199 Two dictionaries: selection and rendering. The local configurations are merged into the global ones. 

200 """ 

201 item_selection_config = ChainMap(config.get("selection", {}), handler_config.get("selection", {})) 

202 item_rendering_config = ChainMap(config.get("rendering", {}), handler_config.get("rendering", {})) 

203 return item_selection_config, item_rendering_config

206class _PostProcessor(Treeprocessor): 

207 def run(self, root: Element): 

208 carry_text = "" 

209 for el in reversed(root): # Reversed mainly for the ability to mutate during iteration. 

210 if el.tag == "div" and el.get("class") == "mkdocstrings": 

211 # Delete the duplicated headings along with their container, but keep the text (i.e. the actual HTML). 

212 carry_text = (el.text or "") + carry_text 

213 root.remove(el) 

214 elif carry_text: 

215 el.tail = (el.tail or "") + carry_text 

216 carry_text = "" 

217 if carry_text: 

218 root.text = (root.text or "") + carry_text
